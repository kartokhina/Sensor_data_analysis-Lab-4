##  Задание
"Анализ данных сенсора"

Вы анализируете данные с температурных датчиков в разных городах.
Каждый ряд — один день (до 30 дней), каждый столбец — измерения за день (до 24 часов)

Программа должна:
. Запросить количество дней d и измерений в дне h;
2. Создать двумерный динамический массив float** data;
3. Считать значения (температуру) с клавиатуры;
4. Реализовать функции:
o float dayAverage(float* row, int h);
o float overallAverage(float** p, int d, int h);
o void showHotDays(float** p, int d, int h); — выводит дни, где средняя температура 
> общей средней;
5. Освободить память.


##  Описание алгоритма
1. Пользователь вводит количество дней и измерений в день
2. Динамически создается двумерный массив нужного размера
3. Заполнение массива температурными данными
4. Вывод всех введенных данных
5. Расчет и вывод средней температуры по каждому дню
6. Расчет и вывод общей средней температуры
7. Поиск и вывод "теплых" дней (выше общей средней)
8. Освобождение выделенной памяти

### Основные функции:

## Основные функции:

1. inputData() - ввод данных в двумерный массив
Запрашивает и считывает значения температуры для каждого дня через указатели.

2. printData() - вывод всех измерений
Отображает все введенные данные в виде таблицы (дни × измерения).

3. dayAverage() - средняя температура за день
Вычисляет среднее арифметическое для одной строки (одного дня) через указатели.

4. overallAverage() - общая средняя температура
Рассчитывает среднюю температуру по всем дням и всем измерениям.

5. showHotDays() - вывод теплых дней
Находит и выводит дни, где средняя температура дня выше общей средней.


##  Код программы

```
#include <iostream>

using namespace std;

void inputData(float** data, int d, int h){
    for (int i = 0; i < d; i++){
        cout << "День " << i + 1 << ": ";
        for (int j = 0; j < h; j++) {
            cin >> *(*(data + i)+j);
        }
        
    }
}
void printData(float** data, int d, int h) {
    cout << "\nВсе данные:" << endl;
    for (int i = 0; i < d; i++) {
        cout << "День " << i + 1 << ": ";
        for (int j = 0; j < h; j++) {
            cout <<  *(*(data + i)+j) << " ";
        }
        cout << endl;
    }
}
float dayAverage(const float* row, int h){
    float sum = 0.0;
     for (int i = 0; i < h; i++){
        sum += *(row+i);
    }
    return sum / h;
}

float overallAverage(float** data, int d, int h) {
    float t = 0.0;
    for (int i=0; i < d; i++){
        for (int j=0; j < h; j++){
            t +=  *(*(data + i)+j);
        }
    }
    return t / (d * h);
}

void showHotDays(float** p, int d, int h){
    float overallAvg = overallAverage(p, d, h);
    for (int i; i < d; i++){
        float dayAvg = dayAverage(*(p + i), h);
        if (dayAvg > overallAvg){
             cout << "День " << i + 1 << " (среднее = " << dayAvg << ")" << endl;
        }

    }
}

int main() {
    cout << "=== Анализ сенсора ===" << endl;
    cout << "Введите количество дней: ";
    int d;
    cin >> d;

    cout << "Введите количество измерений в дне: ";
    int h;
    cin >> h;
    
    float** data = new float* [d];
    for (int i = 0; i < d; i++){
        *(data + i) = new float[h];
    }
    

  
    inputData(data, d, h);
    printData(data, d,  h);

    float avg = overallAverage(data, d, h);
    cout << "\nОбщая средняя температура: " << avg << endl;

    showHotDays(data, d, h);
    


    for (int i = 0; i < d; i++) {
        delete[] *(data+i);
    }
    delete[] data;


    return 0;
}
```
